## [Problem Formulation]

### Hard Constraints
- **Input**: `(81, 256, 256)` — 81 ptychographic diffraction intensity patterns, each 256×256
- **Output**: `(2, 600, 600)` — channel 0 = magnitude, channel 1 = phase of complex object

### Forward Model

$$I_j(\mathbf{k}) = \left|\mathcal{F}\left\{P(\mathbf{r} - \mathbf{R}_j) \cdot O(\mathbf{r})\right\}\right|^2 + n, \quad j = 1, \ldots, 81$$

|------|-------|--------|
| 7 | Wrong probe scaling, normalization | PSNR 8.73, SSIM 0.28 |

**Budget calculation**:

- Main reconstruction: 150 iter × 81 pos × 2 FFTs = 24,300 FFTs

### Step 1: Data Loading and Analytical DC Detection

```python
import numpy as np

data = np.load("input.npy")  # (81, 256, 256)
amplitudes = np.sqrt(np.maximum(data, 0.0))

# Analytical DC detection — no reconstruction needed
avg_pattern = np.mean(data, axis=0)  # (256, 256)

# Check if DC peak is at center (128,128) or corner (0,0)
center_val = avg_pattern[124:132, 124:132].mean()
corner_val = np.mean([avg_pattern[:4, :4].mean(), avg_pattern[:4, -4:].mean(),
                       avg_pattern[-4:, :4].mean(), avg_pattern[-4:, -4:].mean()])

use_fftshift = center_val > corner_val * 2.0

# Also check: find actual peak location for robustness
peak_idx = np.unravel_index(np.argmax(avg_pattern), avg_pattern.shape)
# If peak is near center, definitely use fftshift
if 100 < peak_idx[0] < 156 and 100 < peak_idx[1] < 156:
    use_fftshift = True
elif peak_idx[0] < 30 or peak_idx[0] > 226:
    if peak_idx[1] < 30 or peak_idx[1] > 226:
        use_fftshift = False

mean_total_intensity = np.mean(np.sum(data, axis=(1, 2)))
```

### Step 2: FFT Convention (Fixed Once)

```python
def forward_prop(x):
    if use_fftshift:
        return np.fft.fftshift(np.fft.fft2(np.fft.ifftshift(x)))
    else:
        return np.fft.fft2(x)

def backward_prop(X):
    if use_fftshift:
        return np.fft.fftshift(np.fft.ifft2(np.fft.ifftshift(X)))
    else:
        return np.fft.ifft2(X)
```

### Step 3: Scan Positions (9×9 grid)

```python
N_scan = 9
probe_size = 256
obj_size = 600

scan_step = (obj_size - probe_size) / (N_scan - 1)  # = 43.0

positions = []
for iy in range(N_scan):
    for ix in range(N_scan):
        ry = int(round(iy * scan_step))
        rx = int(round(ix * scan_step))
        ry = min(ry, obj_size - probe_size)
        rx = min(rx, obj_size - probe_size)
```

### Step 4: Probe Initialization

```python
N_pix = probe_size ** 2
obj_init_value = 1.0

Y, X = np.mgrid[-128:128, -128:128]
R2 = (X**2 + Y**2).astype(np.float64)

def make_probe(sigma):
    P = np.exp(-R2 / (2.0 * sigma**2)).astype(np.complex128)
    current_energy = np.sum(np.abs(P)**2)
    target_energy = mean_total_intensity / N_pix  # Since |O_init|² = 1
    P *= np.sqrt(target_energy / current_energy)
    return P
```

### Step 5: Fast Probe Selection (3 candidates × 15 iterations)

**Budget: 3 × 15 × 81 × 2 = 7,290 FFTs**

```python
def quick_reconstruction(P_init, n_iter=15):
    O = np.ones((obj_size, obj_size), dtype=np.complex128)
    P = P_init.copy()
    alpha_val = 0.5
    
    final_err = np.inf
    for ep in range(n_iter):
        order = np.random.permutation(81)
        epoch_err = 0.0
        for idx in range(81):
            j = order[idx]
            ry, rx = positions[j]
            O_patch = O[ry:ry+probe_size, rx:rx+probe_size].copy()
            
            psi = P * O_patch
            Psi = forward_prop(psi)
            mag = np.abs(Psi)
            Psi_c = amplitudes[j] * Psi / (mag + 1e-12)
            epoch_err += np.sum((mag - amplitudes[j])**2)
            
            psi_c = backward_prop(Psi_c)
            dpsi = psi_c - psi
            
            P_abs2 = np.abs(P)**2
            P_max2 = np.max(P_abs2)
            denom_O = (1.0 - alpha_val) * P_abs2 + alpha_val * P_max2
            O[ry:ry+probe_size, rx:rx+probe_size] += np.conj(P) * dpsi / (denom_O + 1e-12)
            
            if ep >= 3:
                O_abs2 = np.abs(O_patch)**2
                O_max2 = np.max(O_abs2)
                denom_P = (1.0 - alpha_val) * O_abs2 + alpha_val * O_max2
                P += np.conj(O_patch) * dpsi / (denom_P + 1e-12)
        
        total_energy = np.sum(amplitudes**2)
        final_err = epoch_err / total_energy
    
    return final_err

# Only 3 candidates
sigma_candidates = [20, 40, 70]
best_sigma = 40
best_err = np.inf

for sigma in sigma_candidates:
    P_test = make_probe(sigma)
    err = quick_reconstruction(P_test, n_iter=15)
    if err < best_err:
        best_err = err
        best_sigma = sigma

print(f"Best sigma: {best_sigma}, error: {best_err:.6f}")
```

### Step 6: Main Reconstruction — rPIE (150 iterations, patience 40)

**Budget: 150 × 81 × 2 = 24,300 FFTs**

```python
P = make_probe(best_sigma)
O = np.ones((obj_size, obj_size), dtype=np.complex128)

best_O = O.copy()
best_P = P.copy()
best_fourier_error = np.inf
best_epoch = 0

N_iter = 150
alpha = 0.5
probe_update_start = 3
patience = 40

error_history = []

for epoch in range(N_iter):
    order = np.random.permutation(81)
    epoch_err = 0.0
    
    for idx in range(81):
        j = order[idx]
        ry, rx = positions[j]
        O_patch = O[ry:ry+probe_size, rx:rx+probe_size].copy()
        
        psi = P * O_patch
        Psi = forward_prop(psi)
        mag = np.abs(Psi)
        
        Psi_c = amplitudes[j] * Psi / (mag + 1e-12)
        epoch_err += np.sum((mag - amplitudes[j])**2)
        
        psi_c = backward_prop(Psi_c)
        dpsi = psi_c - psi
        
        # rPIE object update
        P_abs2 = np.abs(P)**2
        P_max2 = np.max(P_abs2)
        denom_O = (1.0 - alpha) * P_abs2 + alpha * P_max2
        O[ry:ry+probe_size, rx:rx+probe_size] += np.conj(P) * dpsi / (denom_O + 1e-12)
        
        # Probe update
        if epoch >= probe_update_start:
            O_abs2 = np.abs(O_patch)**2
            O_max2 = np.max(O_abs2)
            denom_P = (1.0 - alpha) * O_abs2 + alpha * O_max2
            P += np.conj(O_patch) * dpsi / (denom_P + 1e-12)
    
    total_energy = np.sum(amplitudes**2)
    rel_err = epoch_err / total_energy
    error_history.append(rel_err)
    
    if rel_err < best_fourier_error:
        best_fourier_error = rel_err
        best_O = O.copy()
        best_P = P.copy()
        best_epoch = epoch
    
    if epoch - best_epoch > patience:
        print(f"Early stopping at epoch {epoch}")
        break

O_final = best_O.copy()
```

### Step 7: Light TV Denoising (15 iterations, conservative)

```python
def tv_denoise_2d(img, weight, n_iter=15):
    """Chambolle TV denoising."""
    px = np.zeros_like(img)
    py = np.zeros_like(img)
    tau = 0.125
    
    for _ in range(n_iter):
        div_p = np.zeros_like(img)
        div_p[1:, :] += px[:-1, :]
        div_p[:-1, :] -= px[:-1, :]
        div_p[:, 1:] += py[:, :-1]
        div_p[:, :-1] -= py[:, :-1]
        
        u = img + weight * div_p
        
        gx = np.zeros_like(img)
        gy = np.zeros_like(img)
        gx[:-1, :] = u[1:, :] - u[:-1, :]
        gy[:, :-1] = u[:, 1:] - u[:, :-1]
        
        norm_g = np.sqrt(gx**2 + gy**2)
        denom = 1.0 + tau * norm_g
        px = (px + tau * gx) / denom
        py = (py + tau * gy) / denom
    
    div_p = np.zeros_like(img)
    div_p[1:, :] += px[:-1, :]
    div_p[:-1, :] -= px[:-1, :]
    div_p[:, 1:] += py[:, :-1]
    div_p[:, :-1] -= py[:, :-1]
    
    return img + weight * div_p

mag_raw = np.abs(O_final)
phase_raw = np.angle(O_final)

mag_out = tv_denoise_2d(mag_raw, weight=0.005, n_iter=15)
cos_tv = tv_denoise_2d(np.cos(phase_raw), weight=0.005, n_iter=15)
sin_tv = tv_denoise_2d(np.sin(phase_raw), weight=0.005, n_iter=15)
phase_out = np.arctan2(sin_tv, cos_tv)
```

### Step 8: Phase Referencing via Histogram Peak

```python
# Build illumination weight map
illum_count = np.zeros((obj_size, obj_size), dtype=np.float64)
for j in range(81):
    ry, rx = positions[j]
    illum_count[ry:ry+probe_size, rx:rx+probe_size] += 1.0

nonzero_illum = illum_count[illum_count > 0]
illum_thresh = np.percentile(nonzero_illum, 50) if len(nonzero_illum) > 0 else 1.0
well_lit = illum_count >= illum_thresh

# Phase histogram — substrate is dominant (duty cycle 0.6 → ~64% is substrate)
phase_well = phase_out[well_lit]
hist_phase, bin_edges = np.histogram(phase_well, bins=100, range=(-np.pi, np.pi))
bin_centers = 0.5 * (bin_edges[:-1] + bin_edges[1:])

dominant_idx = np.argmax(hist_phase)
lo = max(0, dominant_idx - 3)
hi = min(len(hist_phase), dominant_idx + 4)
weights = hist_phase[lo:hi].astype(np.float64)
substrate_phase = np.average(bin_centers[lo:hi], weights=weights) if np.sum(weights) > 0 else bin_centers[dominant_idx]

phase_out = phase_out - substrate_phase
phase_out = np.arctan2(np.sin(phase_out), np.cos(phase_out))

# Phase sign check: pads should have positive phase (ΔΦ = 4πh/λ > 0)
mag_well = mag_out[well_lit]
phase_well_ref = phase_out[well_lit]

# Otsu threshold on magnitude to separate pads from substrate
hist_mag, bin_edges_mag = np.histogram(mag_well, bins=50)
bin_centers_mag = 0.5 * (bin_edges_mag[:-1] + bin_edges_mag[1:])

total = np.sum(hist_mag)
sum_total = np.sum(hist_mag * bin_centers_mag)
sum_bg = 0.0
w_bg = 0
max_var = 0.0
threshold_mag = np.median(mag_well)

for i in range(len(hist_mag)):
    w_bg += hist_mag[i]
    if w_bg == 0:
        continue
    w_fg = total - w_bg
    if w_fg == 0:
        break
    sum_bg += hist_mag[i] * bin_centers_mag[i]
    mean_bg = sum_bg / w_bg
    mean_fg = (sum_total - sum_bg) / w_fg
    var_between = w_bg * w_fg * (mean_bg - mean_fg)**2
    if var_between > max_var:
        max_var = var_between
        threshold_mag = bin_centers_mag[i]

class1 = well_lit & (mag_out <= threshold_mag)
class2 = well_lit & (mag_out > threshold_mag)

n1, n2 = np.sum(class1), np.sum(class2)
if n1 >= n2:
    pad_mask = class2
    sub_mask = class1
else:
    pad_mask = class1
    sub_mask = class2

if np.sum(pad_mask) > 50:
    mean_pad_phase = np.mean(phase_out[pad_mask])
    mean_sub_phase = np.mean(phase_out[sub_mask])
    pad_phase_diff = mean_pad_phase - mean_sub_phase
    if pad_phase_diff < -0.1:
        phase_out = -phase_out
```
